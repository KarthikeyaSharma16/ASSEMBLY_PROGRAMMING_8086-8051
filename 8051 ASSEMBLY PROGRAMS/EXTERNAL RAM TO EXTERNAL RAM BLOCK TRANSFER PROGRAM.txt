;WRITE A PROGRAM TO TRANSFER A BLOCK
;OF PROGRAM FROM 5140H TO 9384H IN 
;THE EXTERNAL RAM
;INITIALIZE THE ADDRESS TO BE POINTED
;BY DPTR (SOURCE ADDRESS)
MOV R0, #40H
MOV R1, #51H
;INITIALIZE THE ADDRESS TO BE POINTED
;BY DPTR (DESTINATION ADDRESS)
MOV R2, #84H
MOV R3, #93H
MOV R7, #0AH ;COUNT

BLOCK: MOV DPL, R0
       MOV DPH, R1
       MOVX A, @DPTR
       INC DPTR
       MOV R0, DPL
       MOV R1, DPH
       MOV DPL, R2
       MOV DPH, R3
       MOVX @DPTR, A
       INC DPTR
       MOV R2, DPL
       MOV R3, DPH
       DJNZ R7, BLOCK
